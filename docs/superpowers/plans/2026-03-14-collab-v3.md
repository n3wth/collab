# Collab v3 Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Transform Collab from a single-session demo into a persistent, configurable, multi-user collaborative AI workspace.

**Architecture:** Four independent subsystems built in phases: (1) Supabase persistence for docs/chat/sessions, (2) configurable agent personas replacing hardcoded Aiden/Nova, (3) Yjs networked CRDT via y-websocket for real-time multi-user sync, (4) document template picker with agent context adaptation. Each phase produces working, deployable software.

**Tech Stack:** React 19, Tiptap 3, Yjs + y-websocket, Supabase (Postgres + Realtime), Vite 7, TypeScript 5.9, Vercel (frontend + serverless functions)

```sql
-- Sessions: top-level workspace container
create table sessions (
  id uuid primary key default gen_random_uuid(),
  title text not null default 'Untitled',
  template text not null default 'blank',
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

-- Documents: Yjs binary snapshot per session
create table documents (
  id uuid primary key default gen_random_uuid(),
  session_id uuid not null references sessions(id) on delete cascade,
  yjs_state bytea,
  html_snapshot text,
  updated_at timestamptz not null default now(),
  unique(session_id)
);

-- Chat messages
create table chat_messages (
  id uuid primary key default gen_random_uuid(),
  session_id uuid not null references sessions(id) on delete cascade,
  sender text not null,
  text text not null,
  reasoning jsonb,
  created_at timestamptz not null default now()
);

create index idx_chat_session on chat_messages(session_id, created_at);

-- Agent personas per session
create table agent_personas (
  id uuid primary key default gen_random_uuid(),
  session_id uuid not null references sessions(id) on delete cascade,
  name text not null,
  description text not null,
  system_prompt text not null,
  color text not null default '#1a1a1a',
  owner text not null default 'You',
  model text not null default 'gemini-2.5-flash',
  sort_order int not null default 0
);

create index idx_personas_session on agent_personas(session_id);

-- Enable RLS (public access for now, auth later)
alter table sessions enable row level security;
alter table documents enable row level security;
alter table chat_messages enable row level security;
alter table agent_personas enable row level security;
```

```typescript
// src/lib/session-store.ts
import { supabase } from './supabase'
import type { Session, ChatMessage, AgentPersona, DocTemplate } from '../types'

export async function createSession(title: string, template: DocTemplate): Promise<Session> {
  const { data, error } = await supabase
    .from('sessions')
    .insert({ title, template })
    .select()
    .single()
  if (error) throw error
  return data
}

export async function listSessions(): Promise<Session[]> {
  const { data, error } = await supabase
    .from('sessions')
    .select('*')
    .order('updated_at', { ascending: false })
    .limit(20)
  if (error) throw error
  return data || []
}

export async function getSession(id: string): Promise<Session | null> {
  const { data, error } = await supabase
    .from('sessions')
    .select('*')
    .eq('id', id)
    .single()
  if (error) return null
  return data
}

export async function updateSessionTitle(id: string, title: string): Promise<void> {
  const { error } = await supabase
    .from('sessions')
    .update({ title, updated_at: new Date().toISOString() })
    .eq('id', id)
  if (error) throw error
}

export async function deleteSession(id: string): Promise<void> {
  const { error } = await supabase.from('sessions').delete().eq('id', id)
  if (error) throw error
}

// Document persistence
export async function saveDocument(sessionId: string, html: string): Promise<void> {
  const { error } = await supabase
    .from('documents')
    .upsert({
      session_id: sessionId,
      html_snapshot: html,
      updated_at: new Date().toISOString(),
    }, { onConflict: 'session_id' })
  if (error) throw error
}

export async function loadDocument(sessionId: string): Promise<string | null> {
  const { data, error } = await supabase
    .from('documents')
    .select('html_snapshot')
    .eq('session_id', sessionId)
    .single()
  if (error || !data) return null
  return data.html_snapshot
}

// Chat persistence
export async function saveChatMessage(sessionId: string, msg: Omit<ChatMessage, 'id' | 'session_id' | 'created_at'>): Promise<ChatMessage> {
  const { data, error } = await supabase
    .from('chat_messages')
    .insert({
      session_id: sessionId,
      sender: msg.sender,
      text: msg.text,
      reasoning: msg.reasoning || null,
    })
    .select()
    .single()
  if (error) throw error
  return data
}

export async function loadChatMessages(sessionId: string): Promise<ChatMessage[]> {
  const { data, error } = await supabase
    .from('chat_messages')
    .select('*')
    .eq('session_id', sessionId)
    .order('created_at', { ascending: true })
  if (error) throw error
  return data || []
}

// Agent persona persistence
export async function saveAgentPersonas(sessionId: string, personas: Omit<AgentPersona, 'id' | 'session_id'>[]): Promise<AgentPersona[]> {
  // Delete existing, insert fresh
  await supabase.from('agent_personas').delete().eq('session_id', sessionId)
  const { data, error } = await supabase
    .from('agent_personas')
    .insert(personas.map(p => ({ ...p, session_id: sessionId })))
    .select()
  if (error) throw error
  return data || []
}

export async function loadAgentPersonas(sessionId: string): Promise<AgentPersona[]> {
  const { data, error } = await supabase
    .from('agent_personas')
    .select('*')
    .eq('session_id', sessionId)
    .order('sort_order', { ascending: true })
  if (error) throw error
  return data || []
}
```

```typescript
// src/SessionPicker.tsx
import { useState, useEffect } from 'react'
import { listSessions, createSession, deleteSession } from './lib/session-store'
import type { Session, DocTemplate } from './types'

const TEMPLATES: { value: DocTemplate, label: string }[] = [
  { value: 'blank', label: 'Blank' },
  { value: 'prd', label: 'PRD' },
  { value: 'tech-spec', label: 'Tech Spec' },
  { value: 'meeting-notes', label: 'Meeting Notes' },
]

interface Props {
  onSelect: (session: Session) => void
}

export function SessionPicker({ onSelect }: Props) {
  const [sessions, setSessions] = useState<Session[]>([])
  const [loading, setLoading] = useState(true)
  const [creating, setCreating] = useState(false)
  const [title, setTitle] = useState('')
  const [template, setTemplate] = useState<DocTemplate>('blank')

  useEffect(() => {
    listSessions().then(s => { setSessions(s); setLoading(false) })
  }, [])

  const handleCreate = async () => {
    const session = await createSession(title || 'Untitled', template)
    onSelect(session)
  }

  const handleDelete = async (e: React.MouseEvent, id: string) => {
    e.stopPropagation()
    await deleteSession(id)
    setSessions(s => s.filter(x => x.id !== id))
  }

  if (loading) return <div className="session-picker"><p>Loading...</p></div>

  return (
    <div className="session-picker">
      <div className="session-picker-header">
        <h2>Sessions</h2>
        <button onClick={() => setCreating(!creating)}>New</button>
      </div>

      {creating && (
        <div className="session-create-form">
          <input
            value={title}
            onChange={e => setTitle(e.target.value)}
            placeholder="Session title"
            autoFocus
            onKeyDown={e => e.key === 'Enter' && handleCreate()}
          />
          <div className="template-selector">
            {TEMPLATES.map(t => (
              <button
                key={t.value}
                className={`template-btn ${template === t.value ? 'active' : ''}`}
                onClick={() => setTemplate(t.value)}
              >
                {t.label}
              </button>
            ))}
          </div>
          <button className="create-btn" onClick={handleCreate}>Create</button>
        </div>
      )}

      {sessions.length === 0 && !creating && (
        <p className="session-empty">No sessions yet. Create one to get started.</p>
      )}

      <div className="session-list">
        {sessions.map(s => (
          <div key={s.id} className="session-item" onClick={() => onSelect(s)}>
            <div className="session-item-info">
              <span className="session-item-title">{s.title}</span>
              <span className="session-item-template">{s.template}</span>
            </div>
            <div className="session-item-meta">
              <span className="session-item-date">
                {new Date(s.updated_at).toLocaleDateString()}
              </span>
              <button className="session-item-delete" onClick={e => handleDelete(e, s.id)}>x</button>
            </div>
          </div>
        ))}
      </div>
    </div>
  )
}
```

```typescript
interface Props {

  onChange: (personas: Omit<AgentPersona, 'id' | 'session_id'>[]) => void
}

export function AgentConfigurator({ personas, onChange }: Props) {
  const [editing, setEditing] = useState<number | null>(null)

  const addPreset = (preset: typeof PRESETS[0]) => {
    if (personas.length >= 4) return
    // Avoid duplicate names
    const name = personas.some(p => p.name === preset.name)
      ? `${preset.name} ${personas.length + 1}`
      : preset.name
    onChange([...personas, { ...preset, name, sort_order: personas.length }])
  }

  const remove = (idx: number) => {
    onChange(personas.filter((_, i) => i !== idx))
    if (editing === idx) setEditing(null)
  }

  const update = (idx: number, patch: Partial<Omit<AgentPersona, 'id' | 'session_id'>>) => {
    onChange(personas.map((p, i) => i === idx ? { ...p, ...patch } : p))
  }

    <div className="agent-configurator">
      <div className="agent-config-cards">
        {personas.map((p, i) => (
          <div key={i} className="agent-config-card">
            {editing === i ? (
              <div className="agent-config-edit">
                <input value={p.name} onChange={e => update(i, { name: e.target.value })} placeholder="Name" />
                <textarea value={p.description} onChange={e => update(i, { description: e.target.value })} placeholder="Description" rows={2} />
                <textarea value={p.system_prompt} onChange={e => update(i, { system_prompt: e.target.value })} placeholder="System prompt" rows={3} />
                <input value={p.owner} onChange={e => update(i, { owner: e.target.value })} placeholder="Owner" />
                <button onClick={() => setEditing(null)}>Done</button>
              </div>
            ) : (
              <>
                <div className="agent-config-card-header">
                  <BlobAvatar name={p.name} size={22} />
                  <span className="agent-config-name">{p.name}</span>
                  <span className="agent-config-desc">{p.description}</span>
                </div>
                <div className="agent-config-actions">
                  <button onClick={() => setEditing(i)}>Edit</button>
                  <button onClick={() => remove(i)}>Remove</button>
                </div>
              </>
            )}
          </div>
        ))}
      </div>

      {personas.length < 4 && (
        <div className="agent-presets">
          <span className="agent-presets-label">Add:</span>
          {PRESETS.filter(pr => !personas.some(p => p.name === pr.name)).map(pr => (
            <button key={pr.name} className="agent-preset-btn" onClick={() => addPreset(pr)}>
              {pr.name}
            </button>
          ))}
        </div>
      )}
    </div>
  )
}
```

```typescript
const WS_URL = import.meta.env.VITE_WS_URL || 'ws://localhost:1234'
```
